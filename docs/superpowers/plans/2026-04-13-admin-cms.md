> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a password-protected admin portal accessible from the footer that lets the site owner edit all text copy across every page and manage coupon codes on the deals page, backed by Supabase.

**Architecture:** Static HTML site with Supabase as the backend. A shared `cms.js` script loads on every public page, fetches content from Supabase's `site_content` table, and replaces text in elements marked with `data-cms` attributes. An `admin.html` page provides a password-gated UI for editing all text (grouped by page) and managing coupons. Coupons on the deals page are rendered dynamically from Supabase instead of being hardcoded.

**Tech Stack:** Vanilla JS, Supabase JS Client (CDN), existing CSS variables/classes

- [ ] **Step 4: Create a Supabase Edge Function for authenticated writes**

Since the anon key can only read, we need a serverless function that accepts the admin password and uses the service role to write. In the Supabase Dashboard > Edge Functions, create a function called `admin-write`:

```sql
-- Instead of Edge Functions, use a simpler approach:
-- Create a Postgres function that checks the password and performs writes

CREATE OR REPLACE FUNCTION update_content(
  p_password TEXT,
  p_id TEXT,
  p_content TEXT
) RETURNS BOOLEAN AS $$
DECLARE
  stored_pw TEXT;
BEGIN
  SELECT value INTO stored_pw FROM admin_settings WHERE key = 'admin_password';
  IF stored_pw IS DISTINCT FROM p_password THEN
    RETURN false;
  END IF;
  INSERT INTO site_content (id, content, updated_at)
  VALUES (p_id, p_content, now())
  ON CONFLICT (id) DO UPDATE SET content = p_content, updated_at = now();
  RETURN true;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE OR REPLACE FUNCTION upsert_coupon(
  p_password TEXT,
  p_id UUID,
  p_category TEXT,
  p_type_label TEXT,
  p_code TEXT,
  p_offer TEXT,
  p_detail TEXT,
  p_active BOOLEAN
) RETURNS UUID AS $$
DECLARE
  stored_pw TEXT;
  result_id UUID;
BEGIN
  SELECT value INTO stored_pw FROM admin_settings WHERE key = 'admin_password';
  IF stored_pw IS DISTINCT FROM p_password THEN
    RETURN NULL;
  END IF;
  IF p_id IS NULL THEN
    INSERT INTO coupons (category, type_label, code, offer, detail, active)
    VALUES (p_category, p_type_label, p_code, p_offer, p_detail, p_active)
    RETURNING id INTO result_id;
  ELSE
    UPDATE coupons SET
      category = p_category,
      type_label = p_type_label,
      code = p_code,
      offer = p_offer,
      detail = p_detail,
      active = p_active
    WHERE id = p_id;
    result_id := p_id;
  END IF;
  RETURN result_id;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE OR REPLACE FUNCTION delete_coupon(
  p_password TEXT,
  p_id UUID
) RETURNS BOOLEAN AS $$
DECLARE
  stored_pw TEXT;
BEGIN
  SELECT value INTO stored_pw FROM admin_settings WHERE key = 'admin_password';
  IF stored_pw IS DISTINCT FROM p_password THEN
    RETURN false;
  END IF;
  DELETE FROM coupons WHERE id = p_id;
  RETURN true;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
```

- [ ] **Step 1: Create cms.js with Supabase client init and content loading**

```javascript
(function () {
  'use strict';

  var SUPABASE_URL = 'https://kokusecmajupumitprue.supabase.co';
  var SUPABASE_ANON_KEY = 'eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpc3MiOiJzdXBhYmFzZSIsInJlZiI6Imtva3VzZWNtYWp1cHVtaXRwcnVlIiwicm9sZSI6ImFub24iLCJpYXQiOjE3NzYxMTgxODYsImV4cCI6MjA5MTY5NDE4Nn0.bXBDmhm1nzmjxwCD4HTIh0FE_S64ej3p05uAATauhZk';

  // Lightweight Supabase REST client (no SDK needed)
  function supabaseGet(table, query) {
    var url = SUPABASE_URL + '/rest/v1/' + table + (query || '');
    return fetch(url, {
      headers: {
        'apikey': SUPABASE_ANON_KEY,
        'Authorization': 'Bearer ' + SUPABASE_ANON_KEY
      }
    }).then(function (r) { return r.json(); });
  }

  function supabaseRpc(fnName, params) {
    return fetch(SUPABASE_URL + '/rest/v1/rpc/' + fnName, {
      method: 'POST',
      headers: {
        'apikey': SUPABASE_ANON_KEY,
        'Authorization': 'Bearer ' + SUPABASE_ANON_KEY,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify(params)
    }).then(function (r) { return r.json(); });
  }

  // Replace text in data-cms elements
  function loadContent() {
    var els = document.querySelectorAll('[data-cms]');
    if (!els.length) return;

    supabaseGet('site_content').then(function (rows) {
      if (!Array.isArray(rows)) return;
      var map = {};
      rows.forEach(function (r) { map[r.id] = r.content; });
      els.forEach(function (el) {
        var key = el.getAttribute('data-cms');
        if (map[key] !== undefined) {
          el.textContent = map[key];
        }
      });
    }).catch(function () {
      // Supabase unreachable — hardcoded HTML shows as fallback
    });
  }

  // Render coupons on deals.html
  function loadCoupons() {
    var grid = document.getElementById('coupon-grid-jetski');
    var boatGrid = document.getElementById('coupon-grid-boat');
    var specialGrid = document.getElementById('coupon-grid-special');
    if (!grid && !boatGrid && !specialGrid) return;

    supabaseGet('coupons', '?active=eq.true&order=created_at.asc').then(function (coupons) {
      if (!Array.isArray(coupons)) return;

      var groups = { 'jet-ski': [], 'boat': [], 'special': [] };
      coupons.forEach(function (c) {
        if (groups[c.category]) groups[c.category].push(c);
      });

      if (grid) renderCouponCards(grid, groups['jet-ski']);
      if (boatGrid) renderCouponCards(boatGrid, groups['boat']);
      if (specialGrid) renderCouponCards(specialGrid, groups['special']);
    }).catch(function () {
      // Supabase unreachable — show nothing or keep static fallback
    });
  }

  function renderCouponCards(container, coupons) {
    if (!coupons.length) {
      container.innerHTML = '';
      return;
    }
    var delays = ['d1', 'd2', 'd3'];
    container.innerHTML = coupons.map(function (c, i) {
      var codeRow = '';
      if (c.code) {
        codeRow = '<p class="coupon-card__code-row">' +
          '<span class="coupon-card__code">' + escHtml(c.code) + '</span>' +
          '<button class="coupon-copy-btn" data-code="' + escHtml(c.code) + '" aria-label="Copy code ' + escHtml(c.code) + '">' +
          '<svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><rect x="9" y="9" width="13" height="13" rx="2"/><path d="M5 15H4a2 2 0 01-2-2V4a2 2 0 012-2h9a2 2 0 012 2v1"/></svg>' +
          '<span class="coupon-copy-btn__label">COPY</span></button></p>';
      } else if (c.category === 'special') {
        codeRow = '<p class="coupon-card__code" style="font-size:1.2rem;">' + escHtml(c.type_label) + '</p>';
      }

      var isSpecial = c.category === 'special';
      var cardClass = 'coupon-card reveal ' + delays[i % 3] + (isSpecial ? ' coupon-card--local' : '');
      var maxWidth = (c.category === 'boat' || isSpecial) ? ' style="max-width:600px;"' : '';

      var footerBtns = '';
      if (c.code) {
        footerBtns = '<a class="btn btn--gold coupon-card__book-btn" href="https://fareharbor.com/embeds/book/underthebridgewatersports/?full-items=yes">BOOK &amp; APPLY CODE</a>' +
          '<a class="btn btn--outline btn--on-dark coupon-card__call-btn" href="tel:4106297433">CALL 410-629-RIDE</a>';
      } else if (c.category === 'boat') {
        footerBtns = '<a class="btn btn--gold coupon-card__book-btn" href="tel:4106297433">CALL TO BOOK: 410-629-RIDE</a>';
      } else {
        footerBtns = '<a class="btn btn--gold coupon-card__book-btn" href="tel:4106297433">CALL FOR LOCAL RATE: 410-629-RIDE</a>';
      }

      return '<div class="' + cardClass + '"' + maxWidth + '>' +
        '<div class="coupon-card__body">' +
        '<p class="coupon-card__type">' + escHtml(c.type_label) + '</p>' +
        codeRow +
        '<p class="coupon-card__offer">' + escHtml(c.offer) + '</p>' +
        '<p class="coupon-card__detail">' + escHtml(c.detail) + '</p>' +
        '</div>' +
        '<div class="coupon-card__foot">' + footerBtns + '</div>' +
        '</div>';
    }).join('');

    // Re-bind copy buttons
    container.querySelectorAll('.coupon-copy-btn').forEach(function (btn) {
      btn.addEventListener('click', function () {
        var code = btn.getAttribute('data-code');
        var label = btn.querySelector('.coupon-copy-btn__label');
        navigator.clipboard.writeText(code).then(function () {
          btn.classList.add('copied');
          label.textContent = 'COPIED!';
          setTimeout(function () {
            btn.classList.remove('copied');
            label.textContent = 'COPY';
          }, 2000);
        });
      });
    });

    // Re-trigger reveal observer for new cards
    if ('IntersectionObserver' in window) {
      var obs = new IntersectionObserver(function (entries) {
        entries.forEach(function (e) {
          if (e.isIntersecting) { e.target.classList.add('visible'); obs.unobserve(e.target); }
        });
      }, { threshold: 0.08 });
      container.querySelectorAll('.reveal').forEach(function (el) { obs.observe(el); });
    }
  }

  function escHtml(str) {
    var d = document.createElement('div');
    d.textContent = str;
    return d.innerHTML;
  }

  // Expose for admin page
  window.CMS = {
    supabaseGet: supabaseGet,
    supabaseRpc: supabaseRpc
  };

  // Run on DOM ready
  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', function () { loadContent(); loadCoupons(); });
  } else {
    loadContent();
    loadCoupons();
  }
})();
```
